### Event Listener

$('#fg_tipo_donatore').on('change', function() {
    var tipo = $(this).val();
    
    if (tipo === 'solo_donatore') {
        // Mostra categoria donatore
        $('.fg-categoria-donatore-section').show();
        // Nascondi iscrizione
        $('.fg-iscrizione-section').hide();
    } else {
        // Nascondi categoria donatore

        // Mostra iscrizione
        $('.fg-iscrizione-section').show();
    }
});
